//GET APPOINTMENTS FREE
SELECT slots.slot FROM slots 
    LEFT JOIN appuntamenti 
    ON slots.slot = appuntamenti.ora 
    and appuntamenti.giorno = '2021-02-13'
    WHERE appuntamenti.ora IS NULL ;


//VALIDATE EXAM IF EXIST
select count(*) from subexams join slots where subexams.doctor = slots.doctor and subexams.duration = slots.slotduration  and subexams.subexam = 'ecografia mammaria' and slots.doctor = 1 and slots.slot = '12:00:00';
